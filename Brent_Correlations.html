%matplotlib inline
import Quandl
import matplotlib.pylab
import matplotlib.pyplot as plt  
import pandas as pd

#BRENT
data = Quandl.get("EIA/PET_RBRTE_D", authtoken="QURmz9jjVm1Wh1L51Chx",trim_start="2015-01-01", trim_end="2016-01-01")

oil['Brent'] = oil['Value']
ax = oil.plot(y = 'Brent', figsize=(12,5), color = 'black')
ax.set_title('Closing Prices 2015-2016')
ax.set_ylabel("Price ($)")

fig = a_income.plot( y='Anadarko', ax=ax )

fig = x_income.plot( y='Exxon', ax=ax, color = 'red' )

### HEDGED VS NONHEDGED ###
#Hedged percentages from Wunderlich Analysis 2015

#HEDGED (>70%)
h_tickers = ['GOOG/NASDAQ_PDCE','GOOG/NYSE_HK','GOOG/NYSE_BBG','GOOG/NYSE_PXD','GOOG/NYSE_REN','GOOG/NASDAQ_ROSE']
h_cname = ['pdc','halcon','barrett','pioneer','resolute','rosetta']
h_d= {}

#POORLY HEDGED (<70%)
r_tickers = ['GOOG/NASDAQ_AREX','GOOG/NYSE_CHK','GOOG/NYSE_XEC','GOOG/NYSE_CLR','GOOG/NYSE_EGN','GOOG/NYSE_EOG','GOOG/NYSE_MHR','GOOG/NYSE_DNR','GOOG/NYSE_SD']
r_cname = ['approach','chesapeake','cimarex','cont','energen','eog','magnum','denbury','sandridge']
r_d = {}  

def datapull(ticker,company,dic):
    for i,j in zip(ticker,company):
        n = 0
        pull = Quandl.get(i, authtoken="QURmz9jjVm1Wh1L51Chx", trim_start="2015-01-01", trim_end="2016-01-01")
        hedges = pull['Close']
        dic[j] = hedges

datapull(h_tickers, h_cname, h_d)
hedges = pd.DataFrame.from_dict(h_d)
hedges['havg'] = hedges.mean(axis = 1)

datapull(r_tickers, r_cname, r_d)
risky = pd.DataFrame.from_dict(r_d)
risky['ravg'] = risky.mean(axis = 1)


### CORRELATIONS ALL STUDIES ###
comps = pd.concat([oil['Brent'],ap_income['Apache'],a_income['Anadarko'],c_income['Continental'],v_income['Chevron'],x_income['Exxon']], axis=1)
comps_c = comps.corr(method='pearson')
l1 = ['Brent','Apache','Anadarko','Continental','Chevron','Exxon']

avg_results = pd.concat([oil['Brent'],ep['E&P'],integ['Integrated']], axis = 1)
avg_results_c = avg_results.corr(method='pearson')
l2 = ['Brent','E&P','Integrated']

hedged = pd.concat([oil['Brent'],risky['ravg'],hedges['havg']], axis=1)
hedged_c = hedged.corr(method='pearson')
l3 = ['Brent','Poorly Hedged','Hedged']

def mapcorr(results,labels):
    heat = plt.figure(figsize=(8,8))
    ax1 = heat.add_subplot(111)
    cax = ax1.matshow(results, interpolation='nearest')
    heat.colorbar(cax)
    ax1.set_xticklabels(['']+ labels)
    ax1.set_yticklabels(['']+ labels)
    ax1.set_title('Price Correlations 2015-2016')

mapcorr(comps_c,l1)
mapcorr(avg_results_c,l2)
print('Companies with >70% hedging show significantly less correlation with movement of the Brent oil benchmark.')
mapcorr(hedged_c,l3)

cortest1 = hedged.plot(kind='scatter',x='havg', y='Brent')

cortest2 = hedged.plot(kind='scatter',x='ravg', y='Brent')

cortest2.set_ylim(20,70)

cortest2.set_ylabel('Brent Crude ($)')
